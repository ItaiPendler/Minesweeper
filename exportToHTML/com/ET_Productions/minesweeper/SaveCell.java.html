8    public class SaveCell implements Serializable {

12    
13       public SaveCell(SaveCell c) 
14       { 
15           isMine = c.isMine(); 
16           isFlagged = c.isFlagged(); 
17           isHidden = c.isHidden(); 
18           isPressed = c.isPressed(); 
19           isFinished = c.isFinished(); 
20           number = c.getNumber(); 
21       } 
22    
23       public SaveCell(int d) 
24       { 
25           isMine = false; 
26           isFlagged = false; 
27           isHidden = true; 
28           isPressed = false; 
29           isFinished = false; 
30           number = 0; 
31           makeMine(this, d); 
32       } 
33    
34       public SaveCell() { 
35           isMine = false; 
36           isFlagged = false; 
37           isHidden = true; 
38           isPressed = false; 
39           isFinished = false; 
40           number = 0; 
41       } 
42    
43       public SaveCell(boolean isMine, boolean isFlagged, boolean isHidden, boolean isFinished, boolean isPressed, int number) { 
44           this.isMine = isMine; 
45           this.isFlagged = isFlagged; 
46           this.isHidden = isHidden; 
47           this.isFinished = isFinished; 
48           this.isPressed = isPressed; 
49           this.number = number; 
50           Log.d("number is " + number, "just started to make the cell"); 
51       } 
52    
53       private static void makeMine(SaveCell c, int mp) { 
54           Random r = new Random(); 
55           int p = r.nextInt(100) + 1; 
56           if (p < mp) { 
57               c.setMine(); 
58               c.setNumber(-1); 
59           } 
60       }
