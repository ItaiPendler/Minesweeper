87       public static int getBatteryPercentage(Context context) { 
88    
89           IntentFilter iFilter = new IntentFilter(Intent.ACTION_BATTERY_CHANGED); 
90           Intent batteryStatus = context.registerReceiver(null, iFilter); 
91    
92           int level = batteryStatus != null ? batteryStatus.getIntExtra(BatteryManager.EXTRA_LEVEL, -1) : -1; 
93           int scale = batteryStatus != null ? batteryStatus.getIntExtra(BatteryManager.EXTRA_SCALE, -1) : -1; 
94    
95           float batteryPct = level / (float) scale; 
96    
97           return (int) (batteryPct * 100); 
98       }
