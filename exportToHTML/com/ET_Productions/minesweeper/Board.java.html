7    import android.util.Log; 
8    import android.view.View; 
9     
10   import java.io.Serializable; 
11   import java.util.Random; 
12    
13   public class Board implements Serializable { 
14    
15       private static final int ROW_NUM = 8; 
16       private static final int COLOUMN_NUM = 8; 
17       private SaveCell cells[][]; 
18    
19    
20       public Board(int mp) 
21       { 
22    
23           cells = new SaveCell[ROW_NUM + 1][COLOUMN_NUM + 1]; 
24           Log.d("Board", "just made cells"); 
25    
26    
27           for (int i = 1; i < cells.length; i++) { 
28               for (int j = 1; j < cells[i].length; j++) { 
29                   SaveCell c = new SaveCell(mp); 
30    
31                   cells[i][j] = c; 
32                   Log.d("Board", "loop #" + i); 
33                   Log.d("Board", "loop #2 #" + j); 
34    
35    
36               } 
37           } 
38    
39           Log.d("Board", "finished making cells"); 
40    
41    
42       } 
43    
44       public Board() { 
45           this.cells = new SaveCell[ROW_NUM + 1][COLOUMN_NUM + 1]; 
46       } 
47    
48       public SaveCell[][] getCells() { 
49           return this.cells; 
50       } 
51    
52       public void setCells(SaveCell[][] c) { 
53           this.cells = c; 
54       } 
55    
56       public int arraylength() { 
57           return this.cells.length; 
58       } 
59    
60    
61       public void setNumbers() { 
62           Random r = new Random(); 
63           for (int i = 1; i < cells.length; i++) { 
64               for (int j = 1; j < cells[i].length; j++) { 
65                   int x = r.nextInt(6) + 1; 
66                   cells[i][j].setNumber(x); 
67               } 
68           } 
69       } 
70    
71       public int ROW_NUM() { 
72           return ROW_NUM + 1; 
73       } 
74    
75       public int COLOUMN_NUM() { 
76           return COLOUMN_NUM + 1; 
77       } 
78    
79   }
