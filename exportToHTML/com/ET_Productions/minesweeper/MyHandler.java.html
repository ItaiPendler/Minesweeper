7    import android.os.Bundle; 
8    import android.os.Handler; 
9    import android.os.Message;

12   public class MyHandler extends Handler { 
13    
14       public GameUIActivity guia; 
15       public MyHandler (GameUIActivity gameUIActivity){ 
16           super(); 
17           this.guia = gameUIActivity; 
18       } 
19    
20       @Override 
21       public void handleMessage(Message msg) { 
22   //        Log.d("C: MH, F: handleMessage",""); 
23           Log.d("C: MH, F: handleMessage","just started"); 
24           super.handleMessage(msg); 
25           Log.d("C: MH, F: handleMessage",""); 
26           Bundle data = msg.getData(); 
27           int num = data.getInt("count"); 
28           Log.d("C: MH, F: handleMessage","got int i, it is " + num); 
29           String WINLOSE = data.getString("WINLOSE"); 
30           Log.d("C: MH, F: handleMessage","got the winlsoe " + WINLOSE); 
31           guia.switchFrame(num,WINLOSE); 
32           Log.d("C: MH, F: handleMessage","finished"); 
33       } 
34   }
