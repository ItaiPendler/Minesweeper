168                  openDialog(); 
169                  break;

171                  String fileName = "Screenshot " + user.getDate(); 
172                  File completeFile = gameManager.store(takeScreenshot(), fileName); 
173   
174                  openFile(completeFile); 
175                  break; 
176              case R.id.statisticsTab: 
177                  Intent stats = new Intent(this, Statistics.class); 
178                  user.setTime(String.valueOf(timer.getText())); 
179                  stats.putExtra("user", user);

183                  toggleMusic(); 
184                  break; 
185              case R.id.winGameTabButn: 
186                  finishGame(); 
187                  break; 
188              default: 
189                  break; 
190          } 
191          return super.onOptionsItemSelected(item); 
192      }

220          int action = event.getAction(); 
221          int keyCode = event.getKeyCode(); 
222          switch (keyCode) { 
223              case KeyEvent.KEYCODE_VOLUME_UP: 
224                  if (action == KeyEvent.ACTION_DOWN) { 
225                      if (gameManager.isFlagMode()) { 
226                          gameManager.setFlagMode(false); 
227                          Button butnFlag = (Button) findViewById(R.id.butnflag); 
228                          butnFlag.setBackgroundResource(R.drawable.unflagtile);

231                          Button butnFlag = (Button) findViewById(R.id.butnflag); 
232                          butnFlag.setBackgroundResource(R.drawable.realflaggedtile); 
233                      } 
234                  } 
235   
236                  return true; 
237              case KeyEvent.KEYCODE_VOLUME_DOWN: 
238                  if (action == KeyEvent.ACTION_DOWN) { 
239                      if (gameManager.isFlagMode()) {

495      private int findNumOfMine() { 
496          int mineCounter = 0; 
497          for (int i = 1; i < gameManager.getBoard().getCells().length; i++) { 
498              for (int j = 1; j < gameManager.getBoard().getCells()[i].length; j++) { 
499                  if (gameManager.getBoard().getCells()[i][j].isMine()) { 
500                      mineCounter += 1; 
501                  } 
502              } 
503          } 
504   
505          return mineCounter; 
506      }
